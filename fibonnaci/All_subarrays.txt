vector<vector<int>> findAllSubarrays(vector<int>& nums) {
    int n = nums.size();
    vector<vector<int>> subarrays;

    for (int i = 0; i < n; i++) {
        for (int j = i; j < n; j++) {
            vector<int> subarray;

            for (int k = i; k <= j; k++) {
                subarray.push_back(nums[k]);
            }

            subarrays.push_back(subarray);
        }
    }

    return subarrays;
}
